# Essentials

## Request Body
```ts
...
app.get(
  "/",
  {
    schema: {
      body: Type.Object({
        name: Type.String(),
      }),
    },
  },
  (request, response) => {
    return `hello ${request.body.name}`;
  }
);
...
```

## Path Parameters
```ts
...
app.get(
  "/:id",
  {
    schema: {
      params: Type.Object({
        id: Type.Number(),
      }),
    },
  },
  (request, response) => {
    return `id: ${request.params.id}`;
  }
);
...
```

## Query Parameters
```ts
...
app.get(
  "/",
  {
    schema: {
      querystring: Type.Object({
        name: Type.String(),
      }),
    },
  },
  (request, response) => {
    return `hello ${request.query.name}`;
  }
);
...
```

## Response Status Code
```ts
...
app.get(
  "/",
  (request, response) => {
    response.status(218)
    
    return `status: ${response.statusCode}`;
  }
);
...
```
